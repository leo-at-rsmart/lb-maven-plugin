274     protected void doFieldHack() {
275         for (final Field field : getClass().getDeclaredFields()) {
276             try {
277                 final Field parentField = getDeclaredField(getClass().getSuperclass(), field.getName());
278                 if (parentField != null) {
279                     getLog().debug("Setting " + field.getName() + " in " + parentField.getDeclaringClass().getName() + " to " + field.get(this));
280                     parentField.set(this, field.get(this));
281                 }
282             }
283             catch (Exception e) {
284             }
285         }
286     }

307     @Override
308     public void execute() throws MojoExecutionException, MojoFailureException {
309         changeLogFile = DEFAULT_UPDATE_FILE;
310         try {
311             Method meth = AbstractLiquibaseMojo.class.getDeclaredMethod("processSystemProperties");
312             meth.setAccessible(true);
313             meth.invoke(this);
314         }
315         catch (Exception e) {
316             e.printStackTrace();
317         }
318         super.project = this.project;
319 
320         ClassLoader artifactClassLoader = getMavenArtifactClassLoader();
321         final File[] propertyFiles = getLiquibasePropertiesFiles();
322         
323         // execute change logs on each database
324         for (final File props : propertyFiles) {
325             try {
326                 propertyFile = props.getCanonicalPath();
327                 doFieldHack();
328                 
329                 configureFieldsAndValues(getFileOpener(artifactClassLoader));
330                 
331                 doFieldHack();
332             }
333             catch (Exception e) {
334                 throw new MojoExecutionException(e.getMessage(), e);
335             }

341     @Override
342     protected void performLiquibaseTask(Liquibase liquibase) throws LiquibaseException {
343         super.performLiquibaseTask(liquibase);
344         
345         getLog().info("Tagging the database");
346         rollbackTag = TEST_ROLLBACK_TAG;
347         liquibase.tag(rollbackTag);
348 
349         final File realChangeLogFile = new File(changeLogFile);
350         // If there isn't an update.xml, then make one       
351         if (!realChangeLogFile.exists()) {
352             try {
353                 final Collection<File> changelogs = scanForChangelogs(changeLogSavePath);
354                 generateUpdateLog(realChangeLogFile, changelogs);
355             }
356             catch (Exception e) {
357                 throw new LiquibaseException(e);
358             }
359         }
360 
361         getLog().info("Doing update");
362         liquibase.update(contexts);
363 
364         getLog().info("Doing rollback");
365         liquibase.rollback(rollbackTag, contexts);
366     }

368     /**
369      * Parses a properties file and sets the assocaited fields in the plugin.
370      *
371      * @param propertiesInputStream The input stream which is the Liquibase properties that
372      *                              needs to be parsed.
373      * @throws org.apache.maven.plugin.MojoExecutionException
374      *          If there is a problem parsing
375      *          the file.
376      */
377     protected void parsePropertiesFile(InputStream propertiesInputStream)
378             throws MojoExecutionException {
379         if (propertiesInputStream == null) {
380             throw new MojoExecutionException("Properties file InputStream is null.");
381         }
382         Properties props = new Properties();
383         try {
384             props.load(propertiesInputStream);
385         }
386         catch (IOException e) {
387             throw new MojoExecutionException("Could not load the properties Liquibase file", e);
388         }
389 
390         for (Iterator it = props.keySet().iterator(); it.hasNext();) {
391             String key = null;
392             try {
393                 key = (String) it.next();
394                 Field field = getDeclaredField(this.getClass(), key);
395 
396                 if (propertyFileWillOverride) {
397                     setFieldValue(field, props.get(key).toString());
398                 } else {
399                     if (!isCurrentFieldValueSpecified(field)) {
400                         getLog().debug("  properties file setting value: " + field.getName());
401                         setFieldValue(field, props.get(key).toString());
402                     }
403                 }
404             }
405             catch (Exception e) {
406                 getLog().info("  '" + key + "' in properties file is not being used by this "
407                         + "task.");
408             }
409         }
410     }

412     /**
413      * This method will check to see if the user has specified a value different to that of
414      * the default value. This is not an ideal solution, but should cover most situations in
415      * the use of the plugin.
416      *
417      * @param f The Field to check if a user has specified a value for.
418      * @return <code>true</code> if the user has specified a value.
419      */
420     private boolean isCurrentFieldValueSpecified(Field f) throws IllegalAccessException {
421         Object currentValue = f.get(this);
422         if (currentValue == null) {
423             return false;
424         }
425 
426         Object defaultValue = getDefaultValue(f);
427         if (defaultValue == null) {
428             return currentValue != null;
429         } else {
430             // There is a default value, check to see if the user has selected something other
431             // than the default
432             return !defaultValue.equals(f.get(this));
433         }
434     }

436     private Object getDefaultValue(Field field) throws IllegalAccessException {
437         List<Field> allFields = new ArrayList<Field>();
438         allFields.addAll(Arrays.asList(getClass().getDeclaredFields()));
439         allFields.addAll(Arrays.asList(AbstractLiquibaseMojo.class.getDeclaredFields()));
440 
441         for (Field f : allFields) {
442             if (f.getName().equals(field.getName() + DEFAULT_FIELD_SUFFIX)) {
443                 f.setAccessible(true);
444                 return f.get(this);
445             }
446         }
447         return null;
448     }

451     /**
452      * Recursively searches for the field specified by the fieldName in the class and all
453      * the super classes until it either finds it, or runs out of parents.
454      * @param clazz The Class to start searching from.
455      * @param fieldName The name of the field to retrieve.
456      * @return The {@link Field} identified by the field name.
457      * @throws NoSuchFieldException If the field was not found in the class or any of its
458      * super classes.
459      */
460     protected Field getDeclaredField(Class clazz, String fieldName)
461         throws NoSuchFieldException {
462         getLog().debug("Checking " + clazz.getName() + " for '" + fieldName + "'");
463         try {
464             Field f = clazz.getDeclaredField(fieldName);
465             
466             if (f != null) {
467                 return f;
468             }
469         }
470         catch (Exception e) {
471         }
472         
473         while (clazz.getSuperclass() != null) {        
474             clazz = clazz.getSuperclass();
475             getLog().debug("Checking " + clazz.getName() + " for '" + fieldName + "'");
476             try {
477                 Field f = clazz.getDeclaredField(fieldName);
478                 
479                 if (f != null) {
480                     return f;
481                 }
482             }
483             catch (Exception e) {
484             }
485         }
486 
487         throw new NoSuchFieldException("The field '" + fieldName + "' could not be "
488                                        + "found in the class of any of its parent "
489                                        + "classes.");
490     }

492     private void setFieldValue(Field field, String value) throws IllegalAccessException {
493         if (field.getType().equals(Boolean.class) || field.getType().equals(boolean.class)) {
494             field.set(this, Boolean.valueOf(value));
495         } else {
496             field.set(this, value);
497         }

500     protected Collection<File> scanForChangelogs(final File searchPath) {
501         final Collection<File> retval = new ArrayList<File>();
502         
503         if (searchPath.getName().endsWith("update")) {
504             return Arrays.asList(searchPath.listFiles());
505         }
506         
507         if (searchPath.isDirectory()) {
508             for (final File file : searchPath.listFiles()) {
509                 if (file.isDirectory()) {
510                     retval.addAll(scanForChangelogs(file));
511                 }
512             }
513         }
514         
515         return retval;
516     }

518     protected void generateUpdateLog(final File changeLogFile, final Collection<File> changelogs) throws FileNotFoundException, IOException {
519         changeLogFile.getParentFile().mkdirs();
520 
521         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
522 		DocumentBuilder documentBuilder;
523 		try {
524 			documentBuilder = factory.newDocumentBuilder();
525 		}
526 		catch(ParserConfigurationException e) {
527 			throw new RuntimeException(e);
528 		}
529 		documentBuilder.setEntityResolver(new LiquibaseEntityResolver());
530 
531 		Document doc = documentBuilder.newDocument();
532 		Element changeLogElement = doc.createElementNS(XMLChangeLogSAXParser.getDatabaseChangeLogNameSpace(), "databaseChangeLog");
533 
534 		changeLogElement.setAttribute("xmlns", XMLChangeLogSAXParser.getDatabaseChangeLogNameSpace());
535 		changeLogElement.setAttribute("xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
536 		changeLogElement.setAttribute("xsi:schemaLocation", "http://www.liquibase.org/xml/ns/dbchangelog http://www.liquibase.org/xml/ns/dbchangelog/dbchangelog-"+ XMLChangeLogSAXParser.getSchemaVersion()+ ".xsd");
537 
538 		doc.appendChild(changeLogElement);
539 
540         for (final File changelog : changelogs) {
541             doc.getDocumentElement().appendChild(includeNode(doc, changelog));
542         }
543 
544         new DefaultXmlWriter().write(doc, new FileOutputStream(changeLogFile));
545     }
546 
547     protected Element includeNode(final Document parentChangeLog, final File changelog) throws IOException {
548         final Element retval = parentChangeLog.createElementNS(XMLChangeLogSAXParser.getDatabaseChangeLogNameSpace(), "include");
549         retval.setAttribute("file", changelog.getCanonicalPath());
550         return retval;
551     }
